```python
#!/usr/bin/env python3
"""
Idempotent patch: add Payment-image branch to the n8n finance workflow.

Inserts two nodes on the Payments branch (Route by target main[1]):

  Route by target --main[1]--> [Payment with image?]
    TRUE  --> [Attach proof_url] --> <PAYMENT_POST_NODE>
    FALSE -------------------------> <PAYMENT_POST_NODE>

Refuses to re-apply if 'Payment with image?' already exists.
"""
import json, sys, os

WF_PATH = os.path.join(os.path.dirname(__file__), 'n8n-finance-workflow.json')

# --- tweakable: verify this matches the actual node name found in Task 3.1 ---
PAYMENT_POST_NODE = 'POST to CRM (Payment)'   # <-- replace if Task 3.1 found a different name

IF_NODE_NAME     = 'Payment with image?'
ATTACH_NODE_NAME = 'Attach proof_url'

IF_NODE = {
    'name': IF_NODE_NAME,
    'type': 'n8n-nodes-base.if',
    'typeVersion': 2.2,
    'position': [1800, 600],   # visual only; snap later in UI if needed
    'parameters': {
        'conditions': {
            'options': {'caseSensitive': False, 'typeValidation': 'loose'},
            'combinator': 'and',
            'conditions': [{
                'id': 'has-image',
                'operator': {'type': 'string', 'operation': 'startsWith'},
                'leftValue': "={{ ($('Slack Trigger — new message').item.json.files || [])[0]?.mimetype || '' }}",
                'rightValue': 'image/'
            }]
        }
    }
}

ATTACH_NODE = {
    'name': ATTACH_NODE_NAME,
    'type': 'n8n-nodes-base.code',
    'typeVersion': 2,
    'position': [2000, 500],
    'parameters': {
        'jsCode': (
            "// Attach Slack file permalink as proof_url on the payment payload.\n"
            "const dispatch = $('Dispatch by category').item.json;\n"
            "const files = $('Slack Trigger — new message').item.json.files || [];\n"
            "const firstImage = files.find(f => (f.mimetype || '').startsWith('image/'));\n"
            "if (firstImage) {\n"
            "  dispatch.payload = dispatch.payload || {};\n"
            "  dispatch.payload.proof_url = firstImage.permalink;\n"
            "}\n"
            "return { json: dispatch };\n"
        )
    }
}


def main():
    wf = json.load(open(WF_PATH))

    # Idempotency: if IF node already there, bail.
    names = {n['name'] for n in wf['nodes']}
    if IF_NODE_NAME in names:
        print(f"noop: '{IF_NODE_NAME}' already present")
        return 0
    if PAYMENT_POST_NODE not in names:
        print(f"ERROR: target node '{PAYMENT_POST_NODE}' not found in workflow. "
              "Confirm name via Task 3.1 and update PAYMENT_POST_NODE in this script.",
              file=sys.stderr)
        return 2

    wf['nodes'].append(IF_NODE)
    wf['nodes'].append(ATTACH_NODE)

    # Rewire: Route by target main[1] -> [PAYMENT_POST_NODE]  becomes
    #         Route by target main[1] -> [IF_NODE]
    #           IF true  -> [ATTACH_NODE] -> [PAYMENT_POST_NODE]
    #           IF false -> [PAYMENT_POST_NODE]
    router_conns = wf['connections']['Route by target']['main']
    # Sanity: main[1] should currently go straight to PAYMENT_POST_NODE
    main1 = router_conns[1]
    if not main1 or main1[0]['node'] != PAYMENT_POST_NODE:
        print(f"ERROR: Route by target main[1] does not go to '{PAYMENT_POST_NODE}'. "
              f"Got: {main1}", file=sys.stderr)
        return 3

    # Replace main[1] to point at IF_NODE
    router_conns[1] = [{'node': IF_NODE_NAME, 'type': 'main', 'index': 0}]

    # IF_NODE.main[0] (true) -> ATTACH_NODE -> PAYMENT_POST_NODE
    # IF_NODE.main[1] (false) -> PAYMENT_POST_NODE
    wf['connections'][IF_NODE_NAME] = {
        'main': [
            [{'node': ATTACH_NODE_NAME, 'type': 'main', 'index': 0}],     # true
            [{'node': PAYMENT_POST_NODE, 'type': 'main', 'index': 0}],    # false
        ]
    }
    wf['connections'][ATTACH_NODE_NAME] = {
        'main': [[{'node': PAYMENT_POST_NODE, 'type': 'main', 'index': 0}]]
    }

    with open(WF_PATH, 'w') as f:
        json.dump(wf, f, indent=2)
    print(f"patched: added {IF_NODE_NAME} and {ATTACH_NODE_NAME}; "
          f"rewired Route by target main[1]")
    return 0


if __name__ == '__main__':
    sys.exit(main())
```
